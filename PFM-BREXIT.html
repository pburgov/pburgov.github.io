import re
def remove_commas(line):
    """
      Funcin que devuelve una cadena eliminando las
      comas que puedan existir entre 2 comillas
      :param line: lnea del RDD
      :return string: cadena parseada

    return re.sub(r'(?!(([^"]*"){2})*[^"]*$),', '', line)

# Mtodo para sustituir los valores nulos por 0
def check_null(value):
    if len(str(value)) != 0:
        return int(value)
    return 0

# Mtodo para categorizar el tamao de los distritos
def get_district_size(population):
    value = 7
    if population < 50000:
        value = 1
    elif 50000 <= population < 100000:
        value = 2
    elif 100000 <= population < 200000:
        value = 3
    elif 200000 <= population < 300000:
        value = 4
    elif 300000 <= population < 400000:
        value = 5
    elif 400000 <= population < 500000:
        value = 6
    return int(value)  
    
    
# Mtodo para realizar las transformaciones en el RDD census_rdd
def transform_census_rdd(line):

    elements = []
    elements.append(str(line[1]))
    elements.append(str(line[2]))
    elements.append(get_district_size(int(line[3])))
    
    # Calculamos los totales de posibles votantes agrupados en 3 tramos
    age_20to40 = int(line[8]) + int(line[9]) + int(line[10]) + int(line[11])
    age_40to65 = int(line[12]) + int(line[13]) + int(line[14]) + int(line[15]) + \
                 int(line[16])   
    age_65plus = int(line[17]) + int(line[18]) + int(line[19]) + int(line[20]) + \
                 check_null(line[21]) + check_null(line[22])
    
    # Calculamos el total de poblacin que puede votar
    total = age_20to40 + age_40to65 + age_65plus 
    
    # Y los porcentajes de cada uno de los grupos
    per_20to40 = (age_20to40 / float(total)) * 100
    per_40to65 = (age_40to65 / float(total)) * 100
    per_65plus = (age_65plus / float(total)) * 100
    
    # Calculamos el total de poblacin de 0-15 aos
    age_0to15 = int(line[4]) + int(line[5]) + int(line[6])
    
    # ndice de envejecimiento de la poblacin.
    indice_envejecimiento = (age_65plus/float(age_0to15)) * 100
    
    elements.append(round(per_20to40, 2))
    elements.append(round(per_40to65, 2))
    elements.append(round(per_65plus, 2))
    elements.append(round(indice_envejecimiento, 2))

    return elements

# Mtodo para realizar las transformaciones en el RDD referendum_rdd
def transform_referendum_rdd(line):

    elements = []
    elements.append(str(line[3]))
    elements.append(str(line[2]))
    elements.append(float(line[8]))
    elements.append(float(line[20]))
    elements.append(float(line[19]))
    result = 0.0
    if int(line[12]) > int(line[11]):
        result = 1.0
    elements.append(result)
    
    return elements

De la observacin de la matriz de correlacin podemos tomar diversas ideas a tener en cuenta posteriormente:

  • A excepcin de leave, que representa el porcentaje de votos que optan por la salida de la UE, no hay ninguna variable altamente relacionada con la variable result. Esta relacin es lgica ya que result es funcin de leave. No vamos, por tanto, a prescindir de momento de ninguna de ellas por este motivo.
  • Con la edad de los votantes aumenta tambin la tendencia a votar al Brexit.
  • El voto a la salida disminuye con el aumento de la poblacin del distrito.

  • Resulta curioso como la variable jobseeker, que representa el ratio de gente en bsqueda de empleo est tan poco relacionada con el resultado, sobre todo en un proceso como el Brexit, en el que se ha puesto sobre la mesa continuamente el argumento de la inmigracion y de la prdida de trabajo de los locales en favor de los extranjeros.

5.1 Adecuación de Variables Categóricas, Ensamblado y Flujo de Trabajo ↑ Menú¶

Antes de comenzar con la aplicacin de los modelos, es necesario someter al dataframe a una serie de transformaciones para que puedan ser usados por aquellos.

Para ello nos valemos de un flujo de trabajo o Pipeline, que encadenar los siguientes trabajos:

  • Transformacin de las variables categricas de tipo numrico mediante la funcin OneHotEncoder.

  • Adicin del resto de variables numricas del dataframe.

  • Ensamblaje de todas las variables predictoras en un vector de tipo numrico haciendo uso de la funcin VectorAssembler.

Todos estos pasos, as como la propia aplicacin del Pipeline al dataframe se realizar con la funcin get_transformed_dataframe, definida para tal efecto y que nos permite reutilizar cdigo.

Esta funcin, que nos devuelve el dataframe transformado y en condiciones de ser usado con posterioridad, consta de 3 parmetros:

  • model_df: dataframe a transformar
  • catCols: variables categricas de tipo numrico
  • numCols: restos de variables numricas NO categricas

Aclaracin parmetro catCols:

Las nicas variables categricas de tipo numrico son coast y size. A pesar de ser usadas ambas posteriormente tanto en los modelos NO supervisados como en los supervisados, se pasan como parmetro y no se incluyen directamente en el cuerpo de la funcin get_transformed_dataframe por razones formales.

def get_transformed_dataframe(model_df, catCols, numCols):
    """
    Funcin que nos devuelve el dataframe que se pasa como parmetro
    transformado y listo para ser usado por los modelos posteriores
    :param model_df: dataframe que va a ser transformado
    :param catCols: variables categricas de tipo numrico
    :param numCols: variables numricas no categricas.
    :return df_transformed: dataframe transformado
    """
    # stages en el Pipeline
    stages = [] 
    
    for catCol in catCols:
        # Con OneHotEncoder convertimos variables categricas en Vectores Dispersos 
        encoder = OneHotEncoder(inputCol=catCol, outputCol=catCol+"Dummy")
        # Aadimos el encoder a stages
        stages += [encoder]

    inputCols = map(lambda c: c + "Dummy", catCols) + numCols
    
    # Con el transformador VectorAssembler obtenemos la columna de salida
    # features
    assembler = VectorAssembler(inputCols=inputCols, outputCol="features") 
    
    # Aadimos el assembler a stages
    stages += [assembler]
    
    # Definimos el pipeline con las operaciones en orden
    pipeline = Pipeline(stages=stages)

    # Aplicamos las transformaciones al dataset
    pipelineModel = pipeline.fit(model_df)
    
    # Transformamos el dataframe original
    df_transformed = pipelineModel.transform(model_df)

    # Antes de devolver el dataframe transformado, le 
    # quitamos las columnas dummy.
    for catCol in catCols:
        df_transformed = df_transformed.drop(catCol+"Dummy")

    return df_transformed

from pyspark.ml.evaluation import BinaryClassificationEvaluator
from pyspark.ml.tuning import CrossValidator

# Declaramos globalmente el BinaryClassificationEvaluator    
evaluator = BinaryClassificationEvaluator(labelCol='result')


def get_predictions(estimator, paramGrid):
    """
    Funcin para aplicar una Validacin Cruzada a una 
    clasificacin con BinaryClassificationEvaluator a distintos
    modelos
    :param estimator: modelo al que sometemos a CV
    :paramGrid: parmetros usados por el estimador
    :return predictions, modelo
    """
    # Declaramos el CrossValidator
    cv = CrossValidator(estimator=estimator, 
                        estimatorParamMaps=paramGrid, 
                        evaluator=evaluator, 
                        numFolds=5)
    
    # Creamos el modelo aplicando el algoritmo al conjunto de entrenamiento
    modelo = cv.fit(train)
    
    # Aplicamos el modelo sobre el conjunto de test para obtener las predicciones
    predictions = modelo.transform(test)
    
    return predictions, modelo

from sklearn.metrics import roc_curve
InteractiveShell.ast_node_interactivity = 'last_expr'

palette_roc = ['#006d2c', '#31a354', '#74c476']

def plot_roc_curve(prediction):
    """
    Funcin que nos devuelve los valores de FPR y TPR
    para representar las curvas ROC
    :param prediction: Spark Dataframe con las predicciones
    del modelo
    :return: fpr, tpr
    """
    results = prediction.select(['probability', 'result'])
    results_collect = results.collect()
    results_list = [(float(i[0][0]), 1-float(i[1])) for i in results_collect]
    fpr = dict()
    tpr = dict()
    y_test = [i[1] for i in results_list]
    y_score = [i[0] for i in results_list]
    fpr, tpr, _ = roc_curve(y_test, y_score)
    return fpr, tpr


def representacion_curvas(dataframe):
    """
    Funcin que representa un grfico con las curvas ROC
    de los modelos que se pasan como parmetro mediante
    un Panda Dataframe
    :param: dataframe: Panda Dataframe
    """
    for index, row in dataframe.iterrows():
        fpr, tpr = plot_roc_curve(row['Predictions'])
        plt.plot(fpr, tpr, label='%s (AUC = %0.3f)' % (row['Modelo'] ,row['AUC']),
                color=palette_roc[index])
    plt.plot([0, 1], [0, 1], '--', color='gray')
    plt.xlim([0.0, 1.0])
    plt.ylim([0.0, 1.05])
    ax.set_ylabel('TPR', color=palette[1])
    ax.set_xlabel('FPR', color=palette[1])
    ax.yaxis.set_label_coords(-0.06,0.5)
    ax.xaxis.set_label_coords(0.5, -0.1)
    ax.legend(loc="lower right", fontsize=12)   
    
    
def get_roc_points(dataframe):    
    """
    Funcin que nos devuelve un Panda Dataframe
    con los puntos de la ROC para exportarlos
    posteriormente y representarlos en Tableau
    :param: dataframe: Panda Dataframe
    :return: roc_pd
    """
    roc_pd = pd.DataFrame()
    for index, row in dataframe.iterrows():
        fpr, tpr = plot_roc_curve(row['Predictions'])
        i = len(tpr)
        temp_pd = pd.DataFrame({'Modelo': np.full(i, row['Modelo']),
                                'FPR': pd.Series(fpr),
                                'TPR': pd.Series(tpr)})
        roc_pd =  roc_pd.append(temp_pd)
    return roc_pd
